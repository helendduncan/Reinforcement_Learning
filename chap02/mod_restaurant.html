    op=np.random.normal(m[r],s[r])

def first_visit():
    fv=np.zeros(3)
    nvis=np.zeros(3)
    for _ in range(3):
        fv[_]=visit(_)
        nvis[_] += 1
    best=np.argmax(fv)
    satisfaction=sum(fv)
    return(best, fv, satisfaction, nvis)
def exploit(best, fv, satisfaction, nvis):
